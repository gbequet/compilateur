program p
    var n,f,f1,f2: int;

    begin
        write "nombre de terme a afficher";
        read n;
        f1 := -1;
        f2 := 1;
        while n > -1 do
        begin
          f := f1 + f2;
          f1 := f2;
          f2 := f;
          write f;

          n := n - 1;
        end;
    end
