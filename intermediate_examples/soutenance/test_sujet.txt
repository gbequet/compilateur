program p
    var i,j,maximum : int;
    var t : array[-1..1, 0..1] of int;

    begin
        i := -1;
        j := 0;
        while i <= 1 do
        begin
            while j <= 1 do
            begin
                read t[i,j];
                j := j+1
            end;
            j := 0;
        i := i+1;
        end;

        i := -1;
        j := 0;
        maximum := t[-1,0];
        while i <= 1 do
        begin
            while j <= 1 do
            begin
                if t[i,j] > maximum then maximum := t[i,j];
                j := j+1 
            end;
            j := 0;
        i := i+1; 
        end;

        write "le maximum est";
        write maximum;
    end
